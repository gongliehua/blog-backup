安装 Hexo

# 前提是你已经安装好了nodejs
# 全局安装Hexo
$ npm install hexo-cli -g
# 在当前目录初始化Hexo
$ hexo init
# 安装Hexo所需要的包
$ npm install
# 开启Hexo服务,开启后可以通过localhost:4000查看
$ hexo server

# 安装关于Hexo相关的扩展
$ npm install hexo-deployer-git --save
$ npm install hexo-generator-feed --save
$ npm install hexo-generator-sitemap --save
$ npm install hexo-generator-baidu-sitemap --save

# 基于安装扩展的一些配置信息,添加到配置文件中
deploy:
  type: git
  repository: git@git.coding.net:example/hexo.git
  branch: coding-pages
feed:
  type: atom
  path: atom.xml
  limit: 20
sitemap:
  path: sitemap.xml
baidusitemap:
  path: baidusitemap.xml

# 生成静态文件
$ hexo generate
# 部署到仓库
$ hexo deploy
