Nginx secure_link_module 模块

主要用于资源访问限制。

Syntax: secure_link expression;
Default: -
Content: http,server,location

Syntax: secure_link_md5 expression;
Defult: -
Content:http,server,location

# 例子
location / {
    secure_link $arg_md5,$arg_expires;
    secure_like_md5 "$secure_link_expires$uri imooc";

   if ($secure_link = "") {
        return 403;
   }
   if ($secure_link = 0) {
        return 410;
   }
}
